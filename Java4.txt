import java.util.scanner;
class Java4
{
public static void main(String[] args)
{
Scanner sc=new Scanner(System.in);
System.out.println("enter number:");
int n=sc.nextInt();
for (int i=1;i<=10;i++)
{
System.out.println(n+"* "+i+"="+n*i);
}
}
}
